# Vidar MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the Vidar smart home orchestration MVP — device digital twins via Akka.NET cluster sharding, Shelly HTTP and Zigbee2MQTT integrations as cluster nodes, REST API with SSE, React frontend, Docker Compose deployment.

**Architecture:** Akka.NET cluster with sharded device twin entities. Each device is a GUID-identified sharded actor holding in-memory state and persisting to MongoDB. Communication nodes (Shelly, Zigbee2MQTT) join the cluster as separate processes, translate vendor protocols into typed Akka messages, and route them to twins via sharding. Commands flow back via Distributed Pub/Sub topics. A React SPA served by Kestrel provides configuration and basic controls.

**Tech Stack:** .NET 10, C#, Akka.NET 1.5.68 (Hosting, Cluster, Sharding, Streams, Cluster.Tools), ASP.NET Core, MongoDB.Driver 3.9.0, HiveMQtt 0.44.0, React 19, TypeScript, Vite, Docker Compose, EMQX 5

```csharp
namespace Vidar.Core.Sharding;

public sealed class DeviceTwinMessageExtractor : HashCodeMessageExtractor
```

```csharp
using MongoDB.Driver;
using Vidar.Core.Model;
using Vidar.Host.Persistence;

namespace Vidar.Host.Tests.Persistence;

public class MongoRoomRepositoryTests : IAsyncLifetime
{
    private readonly MongoClient _client;
    private readonly IMongoDatabase _database;
    private readonly MongoRoomRepository _sut;
    private readonly string _dbName = $"vidar_test_{Guid.NewGuid():N}";

    public MongoRoomRepositoryTests()
    {
        var connectionString = Environment.GetEnvironmentVariable("VIDAR_TEST_MONGO")
            ?? "mongodb://localhost:5281";
        _client = new MongoClient(connectionString);
        _database = _client.GetDatabase(_dbName);
        _sut = new MongoRoomRepository(_database);
    }

    public Task InitializeAsync() => Task.CompletedTask;

    public async Task DisposeAsync() => await _client.DropDatabaseAsync(_dbName);

    [Fact]
    public async Task CreateAndGetById_ReturnsRoom()
    {
        var room = new RoomConfiguration { Id = Guid.NewGuid(), Name = "Living Room" };
        await _sut.CreateAsync(room);

        var result = await _sut.GetByIdAsync(room.Id);

        Assert.NotNull(result);
        Assert.Equal("Living Room", result.Name);
    }

    [Fact]
    public async Task GetAll_ReturnsAllRooms()
    {
        await _sut.CreateAsync(new RoomConfiguration { Id = Guid.NewGuid(), Name = "Kitchen" });
        await _sut.CreateAsync(new RoomConfiguration { Id = Guid.NewGuid(), Name = "Bedroom" });

        var result = await _sut.GetAllAsync();

        Assert.Equal(2, result.Count);
    }

    [Fact]
    public async Task Update_ModifiesRoom()
    {
        var room = new RoomConfiguration { Id = Guid.NewGuid(), Name = "Old Name" };
        await _sut.CreateAsync(room);
        room.Name = "New Name";

        await _sut.UpdateAsync(room);
        var result = await _sut.GetByIdAsync(room.Id);

        Assert.Equal("New Name", result!.Name);
    }

    [Fact]
    public async Task Delete_RemovesRoom()
    {
        var room = new RoomConfiguration { Id = Guid.NewGuid(), Name = "Temp" };
        await _sut.CreateAsync(room);

        await _sut.DeleteAsync(room.Id);
        var result = await _sut.GetByIdAsync(room.Id);

        Assert.Null(result);
    }
```

```csharp
using Akka.Actor;
using Akka.Cluster.Tools.PublishSubscribe;
using Akka.Event;
using Vidar.Core.Capabilities;
using Vidar.Core.Messages;
using Vidar.Core.Model;
using Vidar.Host.Persistence;

namespace Vidar.Host.Actors;

public sealed class DeviceTwinActor : ReceiveActor
{
    private readonly ILoggingAdapter _log = Context.GetLogger();
    private readonly string _entityId;
    private readonly IDeviceStateRepository _stateRepo;
    private readonly IDeviceRepository _deviceRepo;
    private DeviceConfiguration? _config;
    private readonly Dictionary<CapabilityType, object> _states = new();

    public static Props Props(string entityId, IDeviceStateRepository stateRepo, IDeviceRepository deviceRepo) =>
        Akka.Actor.Props.Create(() => new DeviceTwinActor(entityId, stateRepo, deviceRepo));

    public DeviceTwinActor(string entityId, IDeviceStateRepository stateRepo, IDeviceRepository deviceRepo)
    {
        _entityId = entityId;
        _stateRepo = stateRepo;
        _deviceRepo = deviceRepo;

        ReceiveAsync<DeviceStateUpdate>(HandleStateUpdate);
        ReceiveAsync<DeviceCommand>(HandleCommand);
    }

    protected override void PreStart()
    {
        base.PreStart();
        LoadConfigAsync().PipeTo(Self);
    }

    private async Task LoadConfigAsync()
    {
        if (Guid.TryParse(_entityId, out var deviceId))
        {
            _config = await _deviceRepo.GetByIdAsync(deviceId);
            var existingState = await _stateRepo.GetByDeviceIdAsync(deviceId);
            if (existingState != null)
            {
                foreach (var kvp in existingState.States)
                    _states[kvp.Key] = kvp.Value;
            }
        }
    }

    private async Task HandleStateUpdate(DeviceStateUpdate update)
    {
        _states[update.Capability] = update.Value;

        var state = new DeviceState
        {
            DeviceId = update.DeviceId,
            States = new Dictionary<CapabilityType, object>(_states),
            LastUpdated = DateTime.UtcNow
        };

        try
        {
            await _stateRepo.UpsertAsync(state);
        }
        catch (Exception ex)
        {
            _log.Warning(ex, "Failed to persist state for device {DeviceId}", update.DeviceId);
        }

        var mediator = DistributedPubSub.Get(Context.System).Mediator;
        mediator.Tell(new Publish(
            "device-state-changes",
            new DeviceStateChanged(update.DeviceId, update.Capability, update.Value, DateTime.UtcNow)));
    }

    private async Task HandleCommand(DeviceCommand command)
    {
        if (_config == null)
        {
            _log.Warning("Received command for unconfigured device {DeviceId}", command.DeviceId);
            return;
        }

        var mediator = DistributedPubSub.Get(Context.System).Mediator;
        mediator.Tell(new Publish($"commands.{_config.CommunicationType}", command));
    }
}
```

```csharp
using System.Threading.Channels;
using Akka.Actor;
using Akka.Cluster.Tools.PublishSubscribe;
using Vidar.Core.Messages;

namespace Vidar.Host.Actors;

public sealed record RegisterSseClient(Channel<DeviceStateChanged> Channel);
public sealed record UnregisterSseClient(Channel<DeviceStateChanged> Channel);

public sealed class SseManagerActor : ReceiveActor
{
    private readonly HashSet<Channel<DeviceStateChanged>> _clients = new();

    public static Props Props() => Akka.Actor.Props.Create(() => new SseManagerActor());

    public SseManagerActor()
    {
        Receive<DeviceStateChanged>(msg =>
        {
            foreach (var client in _clients)
                client.Writer.TryWrite(msg);
        });

        Receive<RegisterSseClient>(msg => _clients.Add(msg.Channel));
        Receive<UnregisterSseClient>(msg =>
        {
            _clients.Remove(msg.Channel);
            msg.Channel.Writer.TryComplete();
        });
    }

    protected override void PreStart()
    {
        var mediator = DistributedPubSub.Get(Context.System).Mediator;
        mediator.Tell(new Subscribe("device-state-changes", Self));
    }
```

```csharp
        await _roomRepo.UpdateAsync(room);
        return NoContent();
    }

    [HttpDelete("{id:guid}")]
    public async Task<IActionResult> Delete(Guid id)
    {

        if (room == null) return NotFound();
        await _roomRepo.DeleteAsync(id);
        return NoContent();

        var response = devices.Select(d => new DeviceResponse(
```

```csharp
using System.Text.Json;
using Akka.Actor;
using Akka.Cluster.Tools.PublishSubscribe;
using Akka.Event;
using Vidar.Core.Capabilities;
using Vidar.Core.Messages;

namespace Vidar.Communication.Shelly;

public sealed record PollDevices;
public sealed record RegisterShellyDevice(ShellyDevice Device);

public sealed class ShellyBridgeActor : ReceiveActor, IWithTimers
{
    private readonly ILoggingAdapter _log = Context.GetLogger();
    private readonly ShellyHttpClient _httpClient;
    private readonly IActorRef _shardRegion;
    private readonly Dictionary<string, ShellyDevice> _devices = new();

    public ITimerScheduler Timers { get; set; } = null!;

    public static Props Props(ShellyHttpClient httpClient, IActorRef shardRegion) =>
        Akka.Actor.Props.Create(() => new ShellyBridgeActor(httpClient, shardRegion));

    public ShellyBridgeActor(ShellyHttpClient httpClient, IActorRef shardRegion)
    {
        _httpClient = httpClient;
        _shardRegion = shardRegion;

        Receive<RegisterShellyDevice>(msg =>
        {
            _devices[msg.Device.NativeId] = msg.Device;
            _log.Info("Registered Shelly device: {NativeId} at {Host}", msg.Device.NativeId, msg.Device.Host);
        });

        ReceiveAsync<PollDevices>(async _ =>
        {
            foreach (var device in _devices.Values)
            {
                if (device.VidarDeviceId == null) continue;
                try
                {
                    var status = await _httpClient.GetStatusAsync(device.Host);
                    var root = status.RootElement;

                    if (root.TryGetProperty("switch:0", out var switchEl))
                    {
                        foreach (var update in ShellyStateMapper.MapSwitchStatus(switchEl))
                            _shardRegion.Tell(new DeviceStateUpdate(device.VidarDeviceId.Value, update.Capability, update.Value));
                    }
                    if (root.TryGetProperty("cover:0", out var coverEl))
                    {
                        foreach (var update in ShellyStateMapper.MapCoverStatus(coverEl))
                            _shardRegion.Tell(new DeviceStateUpdate(device.VidarDeviceId.Value, update.Capability, update.Value));
                    }
                    if (root.TryGetProperty("temperature:0", out var tempEl))
                    {
                        foreach (var update in ShellyStateMapper.MapTemperatureStatus(tempEl))
                            _shardRegion.Tell(new DeviceStateUpdate(device.VidarDeviceId.Value, update.Capability, update.Value));
                    }
                }
                catch (Exception ex)
                {
                    _log.Warning(ex, "Failed to poll Shelly device {NativeId}", device.NativeId);
                }
            }
        });

        Receive<DeviceCommand>(cmd =>
        {
            var device = _devices.Values.FirstOrDefault(d => d.NativeId == cmd.NativeId);
            if (device == null) return;
            _ = ExecuteCommandAsync(device, cmd);
        });
    }

    protected override void PreStart()
    {
        Timers.StartPeriodicTimer("poll", new PollDevices(), TimeSpan.FromSeconds(5), TimeSpan.FromSeconds(10));
        var mediator = DistributedPubSub.Get(Context.System).Mediator;
        mediator.Tell(new Subscribe("commands.shelly", Self));
    }

    private async Task ExecuteCommandAsync(ShellyDevice device, DeviceCommand cmd)
    {
        try
        {
            switch (cmd.Capability)
            {
                case CapabilityType.Switch:
                    await _httpClient.SetSwitchAsync(device.Host, 0, (bool)cmd.Value);
                    break;
                case CapabilityType.Cover:
                    await _httpClient.SetCoverPositionAsync(device.Host, 0, (int)cmd.Value);
                    break;
            }
        }
        catch (Exception ex)
        {
            _log.Warning(ex, "Failed to execute command on {NativeId}", device.NativeId);
        }
    }
}
```

```csharp
using System.Text.Json;
using Vidar.Core.Capabilities;

namespace Vidar.Communication.Zigbee2Mqtt;

public static class ExposesMapper
{
    private static readonly Dictionary<string, CapabilityType> NameToCapability = new()
    {
        ["state"] = CapabilityType.Switch,
        ["brightness"] = CapabilityType.Dimmer,
        ["position"] = CapabilityType.Cover,
        ["temperature"] = CapabilityType.Temperature,
        ["occupancy"] = CapabilityType.Motion,
        ["power"] = CapabilityType.Power,
        ["energy"] = CapabilityType.Energy,
        ["humidity"] = CapabilityType.Humidity,
    };

    public static List<CapabilityType> MapCapabilities(JsonElement exposesArray)
    {
        var capabilities = new HashSet<CapabilityType>();
        foreach (var expose in exposesArray.EnumerateArray())
            CollectCapabilities(expose, capabilities);
        return capabilities.ToList();
    }

    private static void CollectCapabilities(JsonElement expose, HashSet<CapabilityType> capabilities)
    {
        var type = expose.GetProperty("type").GetString()!;

        switch (type)
        {
            case "light":
                capabilities.Add(CapabilityType.Switch);
                if (HasFeature(expose, "brightness"))
                    capabilities.Add(CapabilityType.Dimmer);
                break;

            case "cover":
                capabilities.Add(CapabilityType.Cover);
                break;

            case "binary" or "numeric" or "enum":
                if (expose.TryGetProperty("name", out var nameProp))
                {
                    var name = nameProp.GetString()!;
                    if (NameToCapability.TryGetValue(name, out var capability))
                        capabilities.Add(capability);
                }
                break;
        }

        if (expose.TryGetProperty("features", out var features))
        {
            foreach (var feature in features.EnumerateArray())
                CollectCapabilities(feature, capabilities);
        }
    }

    private static bool HasFeature(JsonElement expose, string featureName) =>
        expose.TryGetProperty("features", out var features) &&
        features.EnumerateArray().Any(f =>
            f.TryGetProperty("name", out var n) && n.GetString() == featureName);
}
```

```csharp
using System.Text.Json;
using Vidar.Core.Capabilities;

namespace Vidar.Communication.Zigbee2Mqtt;

public record Zigbee2MqttCapabilityValue(CapabilityType Capability, object Value);

public static class Zigbee2MqttStateMapper
{
    private static readonly Dictionary<string, CapabilityType> PropertyToCapability = new()
    {
        ["state"] = CapabilityType.Switch,
        ["brightness"] = CapabilityType.Dimmer,
        ["position"] = CapabilityType.Cover,
        ["temperature"] = CapabilityType.Temperature,
        ["occupancy"] = CapabilityType.Motion,
        ["power"] = CapabilityType.Power,
        ["energy"] = CapabilityType.Energy,
        ["humidity"] = CapabilityType.Humidity,
    };

    public static List<Zigbee2MqttCapabilityValue> MapState(string json, IReadOnlyList<CapabilityType> knownCapabilities)
    {
        var doc = JsonDocument.Parse(json);
        var results = new List<Zigbee2MqttCapabilityValue>();

        foreach (var property in doc.RootElement.EnumerateObject())
        {
            if (!PropertyToCapability.TryGetValue(property.Name, out var capability))
                continue;
            if (!knownCapabilities.Contains(capability))
                continue;

            var value = ExtractValue(property.Value, capability);
            if (value != null)
                results.Add(new(capability, value));
        }

        return results;
    }

    private static object? ExtractValue(JsonElement element, CapabilityType capability) => capability switch
    {
        CapabilityType.Switch => element.ValueKind == JsonValueKind.String
            ? element.GetString()!.Equals("ON", StringComparison.OrdinalIgnoreCase)
            : element.GetBoolean(),
        CapabilityType.Motion => element.GetBoolean(),
        CapabilityType.Dimmer or CapabilityType.Cover or CapabilityType.Power
            or CapabilityType.Energy or CapabilityType.Temperature or CapabilityType.Humidity
            => element.GetDouble(),
        _ => null
    };
}
```

```csharp
using System.Text;
using System.Text.Json;
using Akka.Actor;
using Akka.Cluster.Tools.PublishSubscribe;
using Akka.Event;
using HiveMQtt.Client;
using HiveMQtt.Client.Options;
using HiveMQtt.MQTT5.Types;
using Vidar.Core.Capabilities;
using Vidar.Core.Messages;

namespace Vidar.Communication.Zigbee2Mqtt;

public sealed record MqttMessageReceived(string Topic, string Payload);
public sealed record ConnectMqtt;

public sealed class Zigbee2MqttBridgeActor : ReceiveActor
{
    private readonly ILoggingAdapter _log = Context.GetLogger();
    private readonly string _brokerHost;
    private readonly int _brokerPort;
    private readonly IActorRef _shardRegion;
    private HiveMQttClient? _mqttClient;
    private readonly Dictionary<string, Zigbee2MqttDevice> _devices = new();

    public static Props Props(string brokerHost, int brokerPort, IActorRef shardRegion) =>
        Akka.Actor.Props.Create(() => new Zigbee2MqttBridgeActor(brokerHost, brokerPort, shardRegion));

    public Zigbee2MqttBridgeActor(string brokerHost, int brokerPort, IActorRef shardRegion)
    {
        _brokerHost = brokerHost;
        _brokerPort = brokerPort;
        _shardRegion = shardRegion;

        ReceiveAsync<ConnectMqtt>(HandleConnect);
        Receive<MqttMessageReceived>(HandleMqttMessage);
        Receive<DeviceCommand>(HandleCommand);
    }

    protected override void PreStart()
    {
        Self.Tell(new ConnectMqtt());
        var mediator = DistributedPubSub.Get(Context.System).Mediator;
        mediator.Tell(new Subscribe("commands.zigbee2mqtt", Self));
    }

    private async Task HandleConnect(ConnectMqtt _)
    {
        var options = new HiveMQttClientOptions
        {
            Host = _brokerHost,
            Port = _brokerPort,
        };

        _mqttClient = new HiveMQttClient(options);
        var self = Self;

        _mqttClient.OnMessageReceived += (sender, args) =>
        {
            var topic = args.PublishMessage.Topic;
            var payload = Encoding.UTF8.GetString(args.PublishMessage.PayloadAsBytes);
            self.Tell(new MqttMessageReceived(topic!, payload));
        };

        await _mqttClient.ConnectAsync();
        await _mqttClient.SubscribeAsync("zigbee2mqtt/bridge/devices", QualityOfService.AtLeastOnceDelivery);
        await _mqttClient.SubscribeAsync("zigbee2mqtt/bridge/event", QualityOfService.AtLeastOnceDelivery);
        await _mqttClient.SubscribeAsync("zigbee2mqtt/bridge/state", QualityOfService.AtLeastOnceDelivery);
        await _mqttClient.SubscribeAsync("zigbee2mqtt/+", QualityOfService.AtMostOnceDelivery);

        _log.Info("Connected to MQTT broker at {Host}:{Port}", _brokerHost, _brokerPort);
    }

    private void HandleMqttMessage(MqttMessageReceived msg)
    {
        if (msg.Topic == "zigbee2mqtt/bridge/devices")
            HandleDeviceList(msg.Payload);
        else if (msg.Topic == "zigbee2mqtt/bridge/event")
            HandleBridgeEvent(msg.Payload);
        else if (msg.Topic.StartsWith("zigbee2mqtt/") && !msg.Topic.StartsWith("zigbee2mqtt/bridge/"))
            HandleDeviceState(msg.Topic, msg.Payload);
    }

    private void HandleDeviceList(string payload)
    {
        var devices = JsonDocument.Parse(payload).RootElement;
        foreach (var deviceEl in devices.EnumerateArray())
        {
            var type = deviceEl.GetProperty("type").GetString();
            if (type == "Coordinator") continue;

            var ieeeAddress = deviceEl.GetProperty("ieee_address").GetString()!;
            var friendlyName = deviceEl.GetProperty("friendly_name").GetString()!;

            if (_devices.ContainsKey(friendlyName)) continue;

            var capabilities = new List<CapabilityType>();
            if (deviceEl.TryGetProperty("definition", out var def) &&
                def.ValueKind != JsonValueKind.Null &&
                def.TryGetProperty("exposes", out var exposes))
            {
                capabilities = ExposesMapper.MapCapabilities(exposes);
            }

            var metadata = new Dictionary<string, string> { ["ieee_address"] = ieeeAddress };
            if (deviceEl.TryGetProperty("definition", out var def2) && def2.ValueKind != JsonValueKind.Null)
            {
                if (def2.TryGetProperty("model", out var model))
                    metadata["model"] = model.GetString()!;
                if (def2.TryGetProperty("vendor", out var vendor))
                    metadata["vendor"] = vendor.GetString()!;
            }

            var device = new Zigbee2MqttDevice
            {
                IeeeAddress = ieeeAddress,
                FriendlyName = friendlyName,
                Capabilities = capabilities,
                Metadata = metadata
            };
            _devices[friendlyName] = device;

            var discovered = new DeviceDiscovered(
                Guid.NewGuid(), "zigbee2mqtt", ieeeAddress, capabilities, metadata);

            var mediator = DistributedPubSub.Get(Context.System).Mediator;
            mediator.Tell(new Publish("device-discovered", discovered));

            _log.Info("Discovered Zigbee device: {FriendlyName} ({IeeeAddress})", friendlyName, ieeeAddress);
        }
    }

    private void HandleBridgeEvent(string payload)
    {
        var doc = JsonDocument.Parse(payload);
        var eventType = doc.RootElement.GetProperty("type").GetString();
        _log.Debug("Bridge event: {EventType}", eventType);
    }

    private void HandleDeviceState(string topic, string payload)
    {
        var friendlyName = topic.Replace("zigbee2mqtt/", "");
        if (!_devices.TryGetValue(friendlyName, out var device)) return;
        if (device.VidarDeviceId == null) return;

        var updates = Zigbee2MqttStateMapper.MapState(payload, device.Capabilities);
        foreach (var update in updates)
        {
            _shardRegion.Tell(new DeviceStateUpdate(
                device.VidarDeviceId.Value, update.Capability, update.Value));
        }
    }

    private void HandleCommand(DeviceCommand cmd)
    {
        if (_mqttClient == null) return;
        var device = _devices.Values.FirstOrDefault(d => d.IeeeAddress == cmd.NativeId);
        if (device == null) return;

        var payload = BuildCommandPayload(cmd);
        _ = _mqttClient.PublishAsync(
            $"zigbee2mqtt/{device.FriendlyName}/set",
            Encoding.UTF8.GetBytes(payload),
            QualityOfService.AtLeastOnceDelivery);
    }

    private static string BuildCommandPayload(DeviceCommand cmd) => cmd.Capability switch
    {
        CapabilityType.Switch => JsonSerializer.Serialize(new { state = (bool)cmd.Value ? "ON" : "OFF" }),
        CapabilityType.Dimmer => JsonSerializer.Serialize(new { brightness = cmd.Value }),
        CapabilityType.Cover => JsonSerializer.Serialize(new { position = cmd.Value }),
        _ => "{}"
    };
}
```

```tsx
import { useEffect, useState } from 'react';
import { useParams, useNavigate } from 'react-router-dom';
import type { Device } from '../types';
import { api } from '../api/client';
import { connectSSE } from '../api/sse';
import { ToggleSwitch } from '../components/ToggleSwitch';
import { ProgressBar } from '../components/ProgressBar';

export function DeviceDetailPage() {
  const { id } = useParams<{ id: string }>();
  const navigate = useNavigate();
  const [device, setDevice] = useState<Device | null>(null);

  const load = async () => {
    if (!id) return;
    setDevice(await api.devices.get(id));
  };

  useEffect(() => { load(); }, [id]);

  useEffect(() => {
    const sse = connectSSE((event) => {
      if (event.deviceId === id) load();
    });
    return () => sse.close();
  }, [id]);

  if (!device) return <div style={{ color: 'var(--text-dimmed)' }}>Loading...</div>;

  const state = device.state ?? {};

  const handleCommand = async (capability: string, value: unknown) => {
    await api.devices.command(device.id, capability, value);
    await load();
  };

  return (
    <div>
      <button onClick={() => navigate(-1)} style={{ fontSize: 12, color: 'var(--text-dimmed)', marginBottom: 16 }}>
        &larr; Back
      </button>
      <h2 style={{ fontSize: 16, fontWeight: 600, marginBottom: 4 }}>{device.name}</h2>
      <p style={{ fontSize: 12, color: 'var(--text-dimmed)', marginBottom: 16 }}>
        {device.communicationType} — {device.roomName ?? 'No room'}
      </p>

      <div style={{ display: 'flex', flexDirection: 'column', gap: 8 }}>
        {device.capabilities.map(cap => (
          <div key={cap} style={{
            background: 'var(--bg-card)', border: '1px solid var(--border)',
            borderRadius: 6, padding: 12, display: 'flex', justifyContent: 'space-between', alignItems: 'center',
          }}>
            <span style={{ fontSize: 13, color: 'var(--text-secondary)' }}>{cap}</span>
            <div>{renderCapabilityControl(cap, state, handleCommand)}</div>
          </div>
        ))}
      </div>
    </div>
  );
}

function renderCapabilityControl(
  capability: string,
  state: Record<string, unknown>,
  onCommand: (cap: string, val: unknown) => void,
) {
  const value = state[capability];

  switch (capability) {
    case 'Switch':
      return <ToggleSwitch on={value === true} onChange={v => onCommand('Switch', v)} />;
    case 'Dimmer':
      return (
        <input type="range" min={0} max={100} value={(value as number) ?? 0}
          onChange={e => onCommand('Dimmer', Number(e.target.value))}
          style={{ accentColor: 'var(--accent-yellow)' }} />
      );
    case 'Cover':
      return (
        <input type="range" min={0} max={100} value={(value as number) ?? 0}
          onChange={e => onCommand('Cover', Number(e.target.value))}
          style={{ accentColor: 'var(--accent-blue)' }} />
      );
    case 'Temperature':
      return <span style={{ fontSize: 13, color: 'var(--text-muted)' }}>{value ?? '—'}°C</span>;
    case 'Motion':
      return <span style={{ fontSize: 13, color: value ? 'var(--accent-blue)' : 'var(--text-muted)' }}>
        {value ? 'Detected' : 'Clear'}
      </span>;
    case 'Power':
      return <span style={{ fontSize: 13, color: 'var(--text-muted)' }}>{value ?? '—'} W</span>;
    case 'Energy':
      return <span style={{ fontSize: 13, color: 'var(--text-muted)' }}>{value ?? '—'} kWh</span>;
    case 'Humidity':
      return <ProgressBar value={(value as number) ?? 0} label={`${value ?? '—'}%`} />;
    default:
      return <span style={{ fontSize: 13, color: 'var(--text-muted)' }}>{String(value ?? '—')}</span>;
  }
}
```
